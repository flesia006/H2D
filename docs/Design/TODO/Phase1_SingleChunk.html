Task 1.4 — FChunk 데이터 구조

상태: TODO

목적

16³ 블록 컨테이너 + 인덱싱 헬퍼 + dirty/modified 플래그.

코드

// Source/HybridVoxel/Public/Chunk.h
#pragma once
#include "CoreMinimal.h"
#include "VoxelTypes.h"

namespace HybridVoxel {
    constexpr int32 CHUNK_SIZE   = 16;
    constexpr int32 CHUNK_AREA   = CHUNK_SIZE * CHUNK_SIZE;
    constexpr int32 CHUNK_VOLUME = CHUNK_AREA * CHUNK_SIZE;
}

struct HYBRIDVOXEL_API FChunk {
    FIntVector       Coord = FIntVector::ZeroValue;
    TArray<BlockId>  Blocks;  // size = CHUNK_VOLUME
    bool             bDirty    = false;
    bool             bModified = false;

    FChunk() { Blocks.Init(0, HybridVoxel::CHUNK_VOLUME); }

    static FORCEINLINE int32 LocalIndex(int32 X, int32 Y, int32 Z) {
        using namespace HybridVoxel;
        return X + Y * CHUNK_SIZE + Z * CHUNK_AREA;
    }

    FORCEINLINE BlockId Get(int32 X, int32 Y, int32 Z) const {
        return Blocks[LocalIndex(X, Y, Z)];
    }
    FORCEINLINE void Set(int32 X, int32 Y, int32 Z, BlockId Id) {
        Blocks[LocalIndex(X, Y, Z)] = Id;
        bDirty = true; bModified = true;
    }
};
DoD · Get/Set/LocalIndex 단순 호출 테스트로 인덱싱 일치 확인.

Task 1.5 — 페이스 컬링 메시 빌더

상태: TODO

목적

청크 데이터 → 정점/인덱스/UV/노멀 버퍼로 변환. 이웃이 투명인 면만 추가.

코드

// Source/HybridVoxel/Public/ChunkMeshData.h
struct FChunkMeshData {
    TArray<FVector>   Vertices;
    TArray<int32>     Triangles;
    TArray<FVector>   Normals;
    TArray<FVector2D> UV0;
    TArray<FColor>    Colors;       // 바이옴 틴팅 (Phase 4)
};

namespace HybridVoxel {
    // 6면 normal, 4정점 오프셋 (CCW), uv 코너 테이블
    extern const FVector  FaceNormals[6];
    extern const FVector  FaceVerts[6][4]; // 단위 큐브 기준
    extern const FIntVector FaceOffsets[6];
}

HYBRIDVOXEL_API void BuildChunkMesh(const FChunk& Chunk, FChunkMeshData& Out);

// ChunkMesh.cpp (핵심)
void BuildChunkMesh(const FChunk& C, FChunkMeshData& Out) {
    using namespace HybridVoxel;
    const float S = 100.0f; // 블록 한 변 (cm)

    for (int32 z=0; z<CHUNK_SIZE; ++z)
    for (int32 y=0; y<CHUNK_SIZE; ++y)
    for (int32 x=0; x<CHUNK_SIZE; ++x) {
        const BlockId Id = C.Get(x,y,z);
        const FBlockType* Type = FBlockRegistry::Get((EBlockId)Id);
        if (!Type || !Type->bSolid) continue;

        for (int32 f=0; f<6; ++f) {
            FIntVector n = FaceOffsets[f];
            // 청크 경계 밖은 일단 "투명"으로 간주 (Phase 2에서 이웃 청크 조회로 교체)
            const bool bNeighborTransparent = [&](){
                int32 nx=x+n.X, ny=y+n.Y, nz=z+n.Z;
                if (nx<0||ny<0||nz<0||nx>=CHUNK_SIZE||ny>=CHUNK_SIZE||nz>=CHUNK_SIZE) return true;
                const FBlockType* NT = FBlockRegistry::Get((EBlockId)C.Get(nx,ny,nz));
                return !NT || NT->bTransparent || !NT->bSolid;
            }();
            if (!bNeighborTransparent) continue;

            const int32 Base = Out.Vertices.Num();
            const FVector Origin(x*S, y*S, z*S);
            for (int32 v=0; v<4; ++v) {
                Out.Vertices.Add(Origin + FaceVerts[f][v] * S);
                Out.Normals.Add(FaceNormals[f]);
                Out.Colors.Add(FColor::White);
            }
            // UV는 Phase 3 텍스처 아틀라스에서 계산. 일단 0~1
            Out.UV0.Add({0,0}); Out.UV0.Add({1,0});
            Out.UV0.Add({1,1}); Out.UV0.Add({0,1});

            Out.Triangles.Add(Base+0); Out.Triangles.Add(Base+1); Out.Triangles.Add(Base+2);
            Out.Triangles.Add(Base+0); Out.Triangles.Add(Base+2); Out.Triangles.Add(Base+3);
        }
    }
}
DoD · 16³ 풀 청크에서 quad 개수 = 6 × 16 × 16 = 1536 (외곽 면만). 빈 청크는 0 quad.

Task 1.6 — AChunkActor (PMC 래퍼)

상태: TODO

목적

한 청크를 월드에 표현하는 액터. UProceduralMeshComponent 보유 + 메시 적용 API.

코드

// Source/HybridVoxel/Public/ChunkActor.h
#pragma once
#include "GameFramework/Actor.h"
#include "Chunk.h"
#include "ChunkMeshData.h"
#include "ChunkActor.generated.h"

class UProceduralMeshComponent;

UCLASS()
class HYBRIDVOXEL_API AChunkActor : public AActor {
    GENERATED_BODY()
public:
    AChunkActor();

    UPROPERTY(VisibleAnywhere)
    UProceduralMeshComponent* Mesh;

    UPROPERTY(EditAnywhere)
    UMaterialInterface* MasterMaterial = nullptr;

    FIntVector Coord = FIntVector::ZeroValue;

    void ApplyMesh(const FChunkMeshData& D);
};

// ChunkActor.cpp
#include "ChunkActor.h"
#include "ProceduralMeshComponent.h"

AChunkActor::AChunkActor() {
    PrimaryActorTick.bCanEverTick = false;
    Mesh = CreateDefaultSubobject<UProceduralMeshComponent>(TEXT("Mesh"));
    Mesh->bUseAsyncCooking = true;
    SetRootComponent(Mesh);
}

void AChunkActor::ApplyMesh(const FChunkMeshData& D) {
    Mesh->ClearAllMeshSections();
    Mesh->CreateMeshSection_LinearColor(
        /*Section*/ 0,
        D.Vertices, D.Triangles, D.Normals, D.UV0,
        /*VertexColors*/ TArray<FLinearColor>(),
        /*Tangents*/ TArray<FProcMeshTangent>(),
        /*CreateCollision*/ true);
    if (MasterMaterial) Mesh->SetMaterial(0, MasterMaterial);
}
DoD · 레벨에 액터 배치 후 ApplyMesh 호출하면 큐브 군이 렌더 + 캐릭터 충돌.
