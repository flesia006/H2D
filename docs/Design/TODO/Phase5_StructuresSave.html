Task 5.3 — 청크 경계 넘는 스탬핑

상태: TODO

목적

청크 별 확률 판정 → 위치 결정 → 청크 안은 즉시, 밖은 Phase 4의 pending edits 큐 활용.

코드

void TryStampStructures(uint64 Seed, FChunk& C, int32 baseX, int32 baseY, int32 baseZ,
                        const TArray<FStructureTemplate>& Templates,
                        UVoxelWorld& World)
{
    for (const FStructureTemplate& T : Templates) {
        uint64 H = HybridVoxel::DeriveSeed(Seed, C.Coord.X, C.Coord.Y, C.Coord.Z, 4000 + GetTypeHash(T.Name));
        if ((H % 1000) >= (uint64)(T.SpawnRate * 1000.0f)) continue;

        FRandomStream R((int32)H);
        FIntVector Anchor(baseX + R.RandRange(0,15),
                          baseY + R.RandRange(0,15),
                          /*표면 z*/ FindSurfaceZ(C, R.RandRange(0,15), R.RandRange(0,15)));

        for (const FStructureBlock& SB : T.Blocks) {
            FIntVector W = Anchor + SB.Offset;
            FIntVector Cc = HybridVoxel::WorldBlockToChunk(W);
            if (Cc == C.Coord) {
                FIntVector L = HybridVoxel::WorldBlockToLocal(W);
                C.Set(L.X, L.Y, L.Z, (BlockId)SB.Id);
            } else {
                World.EnqueuePendingFor(Cc, { W, (BlockId)SB.Id });
            }
        }
    }
}
DoD · 마을 1채가 두 청크에 걸쳐 있어도 정상 완성. 어느 청크가 먼저 로드돼도 결과 동일.

Task 5.4 — 청크 델타 직렬화

상태: TODO

목적

사용자가 수정한 블록만 (localIndex, blockId) 쌍으로 압축 저장.

코드

struct FBlockEdit { uint16 LocalIndex; uint8 NewId; };

struct FChunkSave {
    FIntVector Coord;
    TArray<FBlockEdit> Edits;

    friend FArchive& operator<<(FArchive& Ar, FChunkSave& S) {
        Ar << S.Coord;
        int32 N = S.Edits.Num(); Ar << N;
        if (Ar.IsLoading()) S.Edits.SetNum(N);
        for (int32 i=0; i<N; ++i) {
            Ar << S.Edits[i].LocalIndex;
            Ar << S.Edits[i].NewId;
        }
        return Ar;
    }
};

// 청크에서 델타 추출
void CollectEdits(const FChunk& C, const FChunk& ProcedurallyGenerated, FChunkSave& Out) {
    Out.Coord = C.Coord;
    for (int32 i=0; i<HybridVoxel::CHUNK_VOLUME; ++i) {
        if (C.Blocks[i] != ProcedurallyGenerated.Blocks[i]) {
            Out.Edits.Add({ (uint16)i, C.Blocks[i] });
        }
    }
}
DoD · 청크당 평균 수십~수백 byte 수준. 수정 0이면 저장 스킵.

// 청크 파일 경로
FString ChunkFilePath(const FString& WorldName, const FIntVector& C) {
    return FPaths::ProjectSavedDir() / TEXT("VoxelWorlds") / WorldName /
           FString::Printf(TEXT("c_%d_%d_%d.bin"), C.X, C.Y, C.Z);
}

void SaveChunkDelta(const FString& WorldName, const FChunkSave& S) {
    TArray<uint8> Bytes;
    FMemoryWriter Ar(Bytes); Ar << const_cast<FChunkSave&>(S);
    FFileHelper::SaveArrayToFile(Bytes, *ChunkFilePath(WorldName, S.Coord));
}

bool LoadChunkDelta(const FString& WorldName, FChunkSave& OutS) {
    TArray<uint8> Bytes;
    if (!FFileHelper::LoadFileToArray(Bytes, *ChunkFilePath(WorldName, OutS.Coord))) return false;
    FMemoryReader Ar(Bytes); Ar << OutS;
    return true;
}
