Task 4.2 — 바이옴 매트릭스 + 표면 치환

상태: TODO

목적

온도/습도 격자로 9개 바이옴 결정. Phase 3 표면 블록(Grass)을 바이옴별로 치환.

코드

enum class EBiome : uint8 {
    Tundra, Taiga, SnowSwamp,
    Plains, Forest, Swamp,
    Desert, Savanna, Jungle
};

EBiome PickBiome(float temp, float hum) {
    int32 t = temp < -0.33f ? 0 : (temp < 0.33f ? 1 : 2);
    int32 h = hum  < -0.33f ? 0 : (hum  < 0.33f ? 1 : 2);
    return (EBiome)(t * 3 + h);
}

EBlockId SurfaceBlockFor(EBiome B) {
    switch (B) {
        case EBiome::Tundra: case EBiome::Taiga: case EBiome::SnowSwamp:
            return EBlockId::Grass; // 임시; 추후 EBlockId::Snow 추가
        case EBiome::Desert: case EBiome::Savanna:
            return EBlockId::Sand;
        default: return EBlockId::Grass;
    }
}
DoD · 한 시드에서 사막/숲/평원이 자연스럽게 인접해서 보임.

Task 4.3 — 3D 동굴 노이즈

상태: TODO

목적

지표 아래 임의 빈 공간 생성. 임계값 튜닝으로 동굴 밀도 조정.

코드

constexpr uint32 SALT_CAVE     = 3001;
constexpr float  CAVE_THRESHOLD = 0.55f;

void CarveCaves(FPerlin& CaveN, FChunk& C, int32 baseX, int32 baseY, int32 baseZ) {
    using namespace HybridVoxel;
    for (int32 z=0; z<CHUNK_SIZE; ++z)
    for (int32 y=0; y<CHUNK_SIZE; ++y)
    for (int32 x=0; x<CHUNK_SIZE; ++x) {
        BlockId& B = C.Blocks[FChunk::LocalIndex(x,y,z)];
        if (B == (BlockId)EBlockId::Air) continue;
        int32 wz = baseZ + z;
        if (wz > 50) continue;  // 표면 가까이는 안 깎음
        float n = CaveN.Noise3D((baseX+x)*0.04f, (baseY+y)*0.04f, wz*0.04f);
        if (n > CAVE_THRESHOLD) B = (BlockId)EBlockId::Air;
    }
}
DoD · 채굴해서 들어가면 자연스러운 동공, 임계값을 0.5~0.65 사이에서 튜닝 가능.

Task 4.4 — 광물 베인 데이터 + 배치

상태: TODO

목적

Stone을 광물로 치환. 깊이별 분포 + 클러스터.

코드

struct FOreVein {
    EBlockId BlockId;
    float    Frequency;   // 0~1, 청크당 시도 비율
    int32    ClusterSize; // 한 베인 평균 블록 수
    int32    ZMin, ZMax;
};

static const FOreVein Veins[] = {
    { EBlockId::CoalOre, 0.04f, 6, 0, 64 },
    { EBlockId::IronOre, 0.02f, 5, 0, 40 },
    { EBlockId::GoldOre, 0.005f,4, 0, 20 },
};

void PlaceOres(uint64 Seed, FChunk& C, int32 baseZ) {
    for (const auto& V : Veins) {
        // 청크 단위 시도 횟수 결정
        FRandomStream R((int32)(HybridVoxel::SplitMix64(Seed ^ (uint64)V.BlockId)));
        int32 tries = (int32)(V.Frequency * 100.0f); // 단순화
        for (int32 t=0; t<tries; ++t) {
            int32 cx = R.RandRange(0, 15);
            int32 cy = R.RandRange(0, 15);
            int32 cz = R.RandRange(0, 15);
            int32 wz = baseZ + cz;
            if (wz < V.ZMin || wz > V.ZMax) continue;
            // 클러스터: 중심에서 짧은 워크
            for (int32 k=0; k<V.ClusterSize; ++k) {
                int32 ox = FMath::Clamp(cx + R.RandRange(-1,1), 0, 15);
                int32 oy = FMath::Clamp(cy + R.RandRange(-1,1), 0, 15);
                int32 oz = FMath::Clamp(cz + R.RandRange(-1,1), 0, 15);
                BlockId& B = C.Blocks[FChunk::LocalIndex(ox,oy,oz)];
                if (B == (BlockId)EBlockId::Stone) B = (BlockId)V.BlockId;
            }
        }
    }
}
DoD · 동굴 벽이나 채굴 갱도에서 광물 베인이 보임. Gold가 Coal보다 명확히 희귀.

Task 4.5 — 식생 (풀, 단순 나무)

상태: TODO

목적

표면 위에 풀(장식 블록) 또는 나무(4~6칸 Wood + Leaves 군) 스폰. 나무는 청크 경계를 넘을 수 있으므로 다음 Task와 연계.

코드

struct FPendingEdit { FIntVector WorldBlock; BlockId Id; };

void GenerateVegetation(uint64 Seed, FChunk& C, EBiome B,
                        int32 baseX, int32 baseY, int32 baseZ,
                        TArray<FPendingEdit>& OutCrossChunk)
{
    if (B == EBiome::Desert || B == EBiome::Tundra) return;

    FRandomStream R((int32)HybridVoxel::SplitMix64(Seed ^ 0xBEE5));
    int32 treeCount = (B == EBiome::Forest ? 4 : 1);
    for (int32 t=0; t<treeCount; ++t) {
        int32 lx = R.RandRange(2, 13);
        int32 ly = R.RandRange(2, 13);
        // 표면 z 찾기 (최상단 Grass)
        int32 sz = -1;
        for (int32 z=15; z>=0; --z) {
            if (C.Get(lx, ly, z) == (BlockId)EBlockId::Grass) { sz = z; break; }
        }
        if (sz < 0 || sz > 12) continue;
        // 줄기
        for (int32 k=1; k<=4; ++k) C.Set(lx, ly, sz+k, (BlockId)EBlockId::Wood);
        // 잎 (이웃 청크로 넘칠 가능성 → OutCrossChunk로 큐잉)
        for (int32 dz=4; dz<=5; ++dz)
        for (int32 dy=-2; dy<=2; ++dy)
        for (int32 dx=-2; dx<=2; ++dx) {
            FIntVector W(baseX+lx+dx, baseY+ly+dy, baseZ+sz+dz);
            // ... 청크 안이면 직접, 밖이면 OutCrossChunk에 push
        }
    }
}
DoD · Forest 바이옴에서 나무 군집 보임. 청크 경계 위의 나무도 잎이 매끄럽게 이어짐.

Task 4.6 — 이웃 청크 pending edits 큐

상태: TODO

목적

아직 생성되지 않은 이웃 청크에 "도착하면 이 블록을 덮어쓰라"는 명령을 큐잉.

코드

// VoxelWorld 내부
TMap<FIntVector, TArray<FPendingEdit>> PendingForChunk;

void EnqueuePendingFor(const FIntVector& ChunkCoord, const FPendingEdit& E) {
    PendingForChunk.FindOrAdd(ChunkCoord).Add(E);
}

// 새 청크 생성 직후
void ApplyPendingEdits(FChunk& C) {
    using namespace HybridVoxel;
    if (auto* List = PendingForChunk.Find(C.Coord)) {
        for (const auto& E : *List) {
            FIntVector L = WorldBlockToLocal(E.WorldBlock);
            C.Set(L.X, L.Y, L.Z, E.Id);
        }
        PendingForChunk.Remove(C.Coord);
    }
}
DoD · 청크 경계에 걸친 나무·구조물(Phase 5) 가지가 빠지지 않음. 청크 로드 순서와 무관.

FColor TintFor(EBiome B, BlockId Id) {
    if (Id != (BlockId)EBlockId::Grass && Id != (BlockId)EBlockId::Leaves)
        return FColor::White;
    switch (B) {
        case EBiome::Jungle:  return FColor(70, 160, 60);
        case EBiome::Savanna: return FColor(180, 180, 80);
        case EBiome::Tundra:  return FColor(200, 230, 220);
        default:              return FColor(120, 200, 90);
    }
}
