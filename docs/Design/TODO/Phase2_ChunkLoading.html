Task 2.2 — 좌표 변환 헬퍼

상태: TODO

목적

월드 블록 좌표 ↔ 청크 좌표 ↔ 청크 내 로컬 좌표 변환을 한곳에 모아 실수 방지.

코드

// VoxelCoord.h
namespace HybridVoxel {

inline FIntVector WorldBlockToChunk(const FIntVector& Wb) {
    auto fdiv = [](int32 v, int32 s){ return (v >= 0) ? (v / s) : ((v - s + 1) / s); };
    return { fdiv(Wb.X, CHUNK_SIZE), fdiv(Wb.Y, CHUNK_SIZE), fdiv(Wb.Z, CHUNK_SIZE) };
}
inline FIntVector WorldBlockToLocal(const FIntVector& Wb) {
    auto fmod = [](int32 v, int32 s){ int32 r = v % s; return (r < 0) ? r + s : r; };
    return { fmod(Wb.X, CHUNK_SIZE), fmod(Wb.Y, CHUNK_SIZE), fmod(Wb.Z, CHUNK_SIZE) };
}
inline FVector ChunkOriginToWorld(const FIntVector& Cc) {
    return FVector(Cc.X * CHUNK_SIZE * 100.0,
                   Cc.Y * CHUNK_SIZE * 100.0,
                   Cc.Z * CHUNK_SIZE * 100.0);
}

} // namespace HybridVoxel
DoD · 음수 좌표(-1, -16, -17 등)에서도 청크/로컬 분해가 일관(chunk*16+local == world).

Task 2.3 — 로드/언로드 큐와 우선순위

상태: TODO

목적

플레이어 인근부터 채우고 멀리 있는 청크는 즉시 파괴하지 않고 캐시 후보로 분리.

코드

void UVoxelWorld::EnqueueAroundPlayer(const FIntVector& PC) {
    const int32 R = LoadRadiusChunks;
    TArray<FIntVector> Wanted;
    for (int32 dz=-R; dz<=R; ++dz)
    for (int32 dy=-R; dy<=R; ++dy)
    for (int32 dx=-R; dx<=R; ++dx)
        Wanted.Add(PC + FIntVector(dx,dy,dz));

    // 가까운 청크 먼저
    Wanted.Sort([&](const FIntVector& A, const FIntVector& B){
        return (A - PC).SizeSquared() < (B - PC).SizeSquared();
    });
    for (const auto& C : Wanted) {
        if (!Chunks.Contains(C) && !Pending.Contains(C)) {
            LoadQueue.Enqueue(C); Pending.Add(C);
        }
    }
}
DoD · 플레이어가 한 청크 경계를 넘으면 새 청크들이 가까운 순으로 큐에 들어감.

Task 2.4 — Tick: 반경 로드 + 언로드

상태: TODO

목적

매 프레임 플레이어 청크 좌표를 파악, 로드 영역 갱신.

코드

void UVoxelWorld::Tick(float DeltaTime, const FVector& PlayerWorld) {
    using namespace HybridVoxel;
    FIntVector PB(FMath::FloorToInt(PlayerWorld.X / 100.0f),
                  FMath::FloorToInt(PlayerWorld.Y / 100.0f),
                  FMath::FloorToInt(PlayerWorld.Z / 100.0f));
    FIntVector PC = WorldBlockToChunk(PB);

    EnqueueAroundPlayer(PC);
    UnloadFarFrom(PC);

    // 프레임당 N개만 처리 (Phase 2.5에서 비동기로 발전)
    for (int32 i=0; i<2; ++i) {
        FIntVector C;
        if (!LoadQueue.Dequeue(C)) break;
        // (임시) 평지 청크 생성 → Phase 3에서 노이즈로 교체
        auto Sp = MakeShared<FChunk>(); Sp->Coord = C;
        if (C.Z < 0) for (int32 i2=0;i2<HybridVoxel::CHUNK_VOLUME;++i2) Sp->Blocks[i2]=(BlockId)EBlockId::Stone;
        Chunks.Add(C, Sp); Pending.Remove(C);
        SpawnOrUpdateActor(C, *Sp);
    }
}
DoD · 플레이어가 이동하면 시야 반경의 청크들이 계속 새로 생기고 멀어진 청크는 사라짐.

// VoxelWorld에서
BlockId UVoxelWorld::GetBlockWorld(const FIntVector& Wb) const {
    using namespace HybridVoxel;
    FIntVector Cc = WorldBlockToChunk(Wb);
    if (const auto* Found = Chunks.Find(Cc)) {
        FIntVector L = WorldBlockToLocal(Wb);
        return (*Found)->Get(L.X, L.Y, L.Z);
    }
    return (BlockId)EBlockId::Air; // 미로드 청크는 일단 Air 취급
}

Task 2.7 — 채굴 / 배치 입력

상태: TODO

목적

플레이어 카메라에서 레이캐스트 → 맞은 블록 좌표 산출 → 좌클릭 파괴 / 우클릭 인접에 배치.

코드

// Player 또는 별도 컴포넌트에서
bool RaycastHitBlock(FIntVector& OutBlock, FIntVector& OutNeighborForPlace) {
    FVector Start = Camera->GetComponentLocation();
    FVector End   = Start + Camera->GetForwardVector() * 600.0f;
    FHitResult Hit;
    if (!GetWorld()->LineTraceSingleByChannel(Hit, Start, End, ECC_Visibility)) return false;

    // 충돌 좌표를 블록 좌표로 변환 (살짝 안쪽으로 -normal*0.5)
    FVector P = Hit.ImpactPoint - Hit.ImpactNormal * 0.5f;
    OutBlock = FIntVector(FMath::FloorToInt(P.X/100), FMath::FloorToInt(P.Y/100), FMath::FloorToInt(P.Z/100));
    FVector Q = Hit.ImpactPoint + Hit.ImpactNormal * 0.5f;
    OutNeighborForPlace = FIntVector(FMath::FloorToInt(Q.X/100), FMath::FloorToInt(Q.Y/100), FMath::FloorToInt(Q.Z/100));
    return true;
}

void OnPrimary() {  // 채굴
    FIntVector B, N; if (RaycastHitBlock(B,N)) World->SetBlockWorld(B, (BlockId)EBlockId::Air);
}
void OnSecondary(EBlockId Selected) {  // 배치
    FIntVector B, N; if (RaycastHitBlock(B,N)) World->SetBlockWorld(N, (BlockId)Selected);
}
DoD · 좌클릭으로 면 한 칸 사라지고 우클릭으로 그 자리 인접에 블록이 생긴다. 청크 메시 즉시 갱신.

Task 2.8 — Dirty 청크 재메싱 스로틀

상태: TODO

목적

블록 변경 시 dirty 표시 → Tick에서 프레임당 N개씩 재메싱 (비동기 권장).

코드

void UVoxelWorld::Tick(float Dt, const FVector& PlayerWorld) {
    // ... 로드 처리 ...

    int32 Budget = 2;
    for (auto& KV : Chunks) {
        if (Budget == 0) break;
        if (!KV.Value->bDirty) continue;
        KV.Value->bDirty = false;
        // 비동기 재메싱 큐잉 (실패 시 동기 fallback)
        EnqueueRemesh(KV.Key);
        --Budget;
    }
}

// SetBlockWorld 시
void UVoxelWorld::SetBlockWorld(const FIntVector& Wb, BlockId Id) {
    auto Cc = HybridVoxel::WorldBlockToChunk(Wb);
    auto L  = HybridVoxel::WorldBlockToLocal(Wb);
    if (auto* C = Chunks.Find(Cc)) {
        (*C)->Set(L.X, L.Y, L.Z, Id);
        // 경계 블록이면 이웃 청크도 dirty
        for (int32 a=0; a<3; ++a) if (L[a]==0 || L[a]==HybridVoxel::CHUNK_SIZE-1) {
            FIntVector N = Cc; N[a] += (L[a]==0 ? -1 : 1);
            if (auto* Nc = Chunks.Find(N)) (*Nc)->bDirty = true;
        }
    }
}
DoD · 연속으로 블록을 깨도 hitch 없음. 경계 블록 파괴 시 이웃 청크 메시도 같이 갱신.
