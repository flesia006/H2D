Task 3.1 — 시드 파생 해시

상태: TODO

목적

월드 시드 + 좌표 + 레이어 솔트로 결정론적 해시 생성. 모든 노이즈 레이어의 기반.

코드

// VoxelHash.h
namespace HybridVoxel {

FORCEINLINE uint64 SplitMix64(uint64 x) {
    x += 0x9E3779B97F4A7C15ull;
    x = (x ^ (x >> 30)) * 0xBF58476D1CE4E5B9ull;
    x = (x ^ (x >> 27)) * 0x94D049BB133111EBull;
    return x ^ (x >> 31);
}

FORCEINLINE uint64 DeriveSeed(int64 WorldSeed, int32 X, int32 Y, int32 Z, uint32 LayerSalt) {
    uint64 H = (uint64)WorldSeed ^ ((uint64)LayerSalt * 0xD2B74407B1CE6E93ull);
    H = SplitMix64(H ^ (uint64)(uint32)X);
    H = SplitMix64(H ^ (uint64)(uint32)Y);
    H = SplitMix64(H ^ (uint64)(uint32)Z);
    return H;
}

} // namespace HybridVoxel
DoD · (seed,X,Y,Z,salt) 같으면 결과 동일. salt만 바꿔도 분포가 시각적으로 무관해 보이는 정도로 분산.

Task 3.3 — 높이맵 함수 (3-옥타브)

상태: TODO

목적

대륙(저주파) + 언덕(중주파) + 디테일(고주파)로 자연스러운 지형 라인.

코드

// TerrainGen.h
namespace HybridVoxel {

constexpr uint32 SALT_CONTINENT = 1001;
constexpr uint32 SALT_HILL      = 1002;
constexpr uint32 SALT_DETAIL    = 1003;
constexpr float  HEIGHT_MIN     = 32.0f;
constexpr float  HEIGHT_MAX     = 96.0f;

float SurfaceHeight(const FPerlin& ContinentN,
                    const FPerlin& HillN,
                    const FPerlin& DetailN,
                    int32 wx, int32 wy)
{
    float c = ContinentN.Noise2D(wx * 0.001f, wy * 0.001f);     // -1..1
    float h = HillN.Noise2D     (wx * 0.01f,  wy * 0.01f) * 0.5f;
    float d = DetailN.Noise2D   (wx * 0.05f,  wy * 0.05f) * 0.1f;
    float u = FMath::Clamp((c + h + d) * 0.5f + 0.5f, 0.f, 1.f);
    return FMath::Lerp(HEIGHT_MIN, HEIGHT_MAX, u);
}

} // namespace HybridVoxel
DoD · 1024×1024 샘플링하여 회색 PNG로 덤프했을 때 자연스러운 등고선 보임.

Task 3.4 — 청크 생성 파이프라인 (Terrain 단계)

상태: TODO

목적

Phase 2의 임시 평지 생성 함수를 노이즈 기반으로 교체.

코드

void GenerateChunkData(int64 Seed, FChunk& C) {
    using namespace HybridVoxel;
    static FPerlin Continent(DeriveSeed(Seed,0,0,0,SALT_CONTINENT));
    static FPerlin Hill     (DeriveSeed(Seed,0,0,0,SALT_HILL));
    static FPerlin Detail   (DeriveSeed(Seed,0,0,0,SALT_DETAIL));

    const int32 baseX = C.Coord.X * CHUNK_SIZE;
    const int32 baseY = C.Coord.Y * CHUNK_SIZE;
    const int32 baseZ = C.Coord.Z * CHUNK_SIZE;

    for (int32 y=0; y<CHUNK_SIZE; ++y)
    for (int32 x=0; x<CHUNK_SIZE; ++x) {
        int32 wx = baseX + x, wy = baseY + y;
        int32 surf = (int32)SurfaceHeight(Continent, Hill, Detail, wx, wy);

        for (int32 z=0; z<CHUNK_SIZE; ++z) {
            int32 wz = baseZ + z;
            BlockId Id = (BlockId)EBlockId::Air;
            if      (wz < surf - 4) Id = (BlockId)EBlockId::Stone;
            else if (wz < surf)     Id = (BlockId)EBlockId::Dirt;
            else if (wz == surf)    Id = (BlockId)EBlockId::Grass;
            C.Blocks[FChunk::LocalIndex(x,y,z)] = Id;
        }
    }
}
DoD · 인게임에서 무한 평원·언덕이 자연스럽게 이어짐. 절벽·구멍 없음.

코드

// 메시 빌드에서 atlas index → uv 사각형 계산
FORCEINLINE void AtlasUV(int32 atlasIdx, FVector2D Out[4]) {
    const int32 cols = 16;
    int32 col = atlasIdx % cols;
    int32 row = atlasIdx / cols;
    float u0 = (col + 0.001f) / cols, u1 = (col + 0.999f) / cols;
    float v0 = (row + 0.001f) / cols, v1 = (row + 0.999f) / cols;
    Out[0] = {u0, v0}; Out[1] = {u1, v0}; Out[2] = {u1, v1}; Out[3] = {u0, v1};
}

// 면 emit 시
const FBlockType* Type = FBlockRegistry::Get((EBlockId)Id);
FVector2D uv[4]; AtlasUV(Type->AtlasIndex[f], uv);
Out.UV0.Add(uv[0]); Out.UV0.Add(uv[1]); Out.UV0.Add(uv[2]); Out.UV0.Add(uv[3]);

코드

// 자동화 테스트(또는 콘솔 명령)
void TestSeedDeterminism() {
    int64 S = 12345;
    FChunk A; A.Coord = {3,7,1}; GenerateChunkData(S, A);
    FChunk B; B.Coord = {3,7,1}; GenerateChunkData(S, B);
    check(A.Blocks == B.Blocks);  // FArray operator== 가정

    FChunk C; C.Coord = {3,7,1}; GenerateChunkData(S + 1, C);
    check(!(A.Blocks == C.Blocks));
}
